Solution of algebraic and transcendental equations:


                1. bisection method:

                deff('y=f(x)','y=x^3-2*x-5');
                a=2, b=3;
                d=0.00001;
                c=1;
                printf('{BISECTION METHOD}successive approximation \na\tb\tm\tf(m)\n');
                while abs(a-b)>d
                    m = (a+b)/2;
                    printf('%f\t%f\t%f\t%f\n',a,b,m,f(m));
                    if f(m)*f(a)>0
                        a=m;
                    else 
                        b=m;
                end
                c= c+1;
                end
                printf('the solution after %i iterations is %d',c,m);

                2. regula falsi method:

                deff('y=f(x)','y = x^3-2*x-5');
                a = 2, b = 3;
                d = 0.01;
                c = 1;
                printf('{REGULA FALSI METHOD}successive approximation :- \na\tb\tm\tf(m)\t\n');
                for i = 1:25
                    x1= (a*f(b)-b*f(a))/(f(b)-f(a));
                    if(f(x1)*f(a))>0
                        b=x1;
                    else
                        a=x1;
                    end
                    if abs(f(x1))< d
                        break;
                    end
                    printf('%f\t%f\t%f\t%f\t%f\n',a,b,f(a),f(b),x1);
                    c = c+1;
                end
                printf('the solution of equation after %i iteration upto 2 decimal places is %g',c,m);

                3.newton raphson method:

                deff('y = f(x)','y = x*exp(x)-1');
                deff('y1 = f1(x)','y1 = (x+1)*exp(x)');
                x0 = 1;
                d = 0.0001;
                c = 0;
                n = 1;
                printf('{NEWTON RAPHSON METHOD}successive iterations :- \nx0\tf(x0)\tf1(x0)\n');
                while n==1
                        x2 = x0;
                        x1 = x0-(f(x0)/f1(x0));
                        x0 = x1;
                        printf('%f\t%f\t%f\n',x2,f(x1),f(x2));
                        c = c+1;
                        if abs(f(x0))< d
                            break;
                        end
                end
                printf('the root of %i iteration is : %f' ,c, x0);

                4.secant method:

                deff('y=f(x)','y=x^3-2*x-5');
                x1=2;
                x2=3;
                n=1;
                c=0;
                printf('{SECANT METHOD} successive iterations:-\nx1\tx2\tx3\tf(x3)\n');
                while n==1
                    x3=(x1*f(x2)-x2*f(x1))/(f(x2)-f(x1));
                    printf('%f\t%f\t%f\t%f\n',x1,x2,x3,f(x3));
                    if (f(x3)*f(x1))>0
                        x2 = x3;
                    else
                        x1 = x3;
                    end
                    if abs (f(x3))< 0.0001
                        break;
                    end
                    c = c+1;
                end
                printf('The root of the equation after %i iteration is : %f',c,x3);

                1.newton's forward difference intercollation:

                x = [10 20 30 40 50];
                y = [46 66 81 93 101];
                n = length(x);
                del = %nan*ones(n,5);
                del(:,1) = y;
                for j = 1:4
                    for i = n:-1:j+1
                        del(i,j+1) = del(i,j) - del(i-1,j);
                    end
                end
                del = round(del*10^3)/10^3;
                mprintf("%5s%5s%7s%7s%7s%7s\n",'x','y','dy','d2y','d3y','d4y');
                disp([x' del])
                del(:,1) = [];

                h = x(2) - x(1);
                p = (X - x(n)) / h;
                x0 = x(n);
                y0 = y(n);
                dely0 = del(n,:);
                Y = y0;
                for i = 1:length(dely0)
                    t = 1;
                    for j = 1:i
                        t = t * (p + j - 1);
                    end
                    Y = Y + t * dely0(i) / factorial(i);
                end

Numerical integration: 


                1. trapezoidal rule:

                x = [0 0.2 0.4 0.6 0.8 1];
                y = [1 1.0857 1.1448 1.1790 1.1891 1.1755];
                h = 0.2;
                n = length(x);
                area = 0;
                for i = 1:n
                    if i==1 | i==n then
                        area = area + y(i);
                    else
                        area = area + 2*y(i);
                    end
                end
                area = area*(h/2);
                printf('the value of integration by trapezoidal rule is : %f',area);

                2.simpson's 1/3 rule:

                x = [0 0.25 0.50 0.75 1];
                y = [0 0.0615 0.2222 0.3956 0.5];
                h = 0.25;
                n = length(x);
                area = 0;
                for i = 1:n
                    if i == 1 | i == n then
                        area = area + y(i);
                    elseif (modulo(i,2))==0 then 
                        area = area + 2*y(i);
                    else
                        area = area + 4*y(i);
                    end
                end
                area = area *(h/3);
                printf('value of integration by simpsons 1/3rd rule is :%f ', area);

                3.simpson's 3/8th rule:

                x = [0 1 2 3 4 5 6];
                y = [1 0.5 0.2 0.1 0.0588 0.0385 0.0270];
                h = 1;
                n = length(x);
                area = 0;
                for i = 1:n
                    if i == 1 | i == n then
                        area = area + y(i);
                    elseif (modulo(i-1,3))==0 then
                        area = area + 2*y(i);
                    else
                        area = area + 3*y(i);
                    end
                end
                area = area*(3*h/8);

Solving linear system of equation using iterative methods.


                1.gauss siedel method:

                a = [27 6 -1; 6 15 2; 1 1 54];
                b = [85 72 110];
                n = length(b);
                tol = 0.0001;
                iter = 1;
                maxit = 5;
                x = zeros(n, 1);
                E = ones(n, 1);
                s = diag(diag(a));
                t = s-a;
                xold = x;
                while (1)
                    for i = 1:n
                        x(i, iter+1) = (b(i)+t(i,:)*xold)/a(i,i);
                        E(i , iter+1) = (x(i, iter+1)-xold(i))/x(i, iter+1)*100;
                        xold(i) = x(i , iter+1);  
                end
                if x(:, iter)==0 then
                    E = 1;
                else  E = sqrt((sum((E(:,iter+1))^2))/n);
                end
                if E <= tol | iter == maxit then
                    break;
                end
                iter = iter +1;
                end
                X = x(:, iter);
                x= round(x*10^5)/10^5;
                x(:,1)=[];
                mprintf("%s%3s%11s%10s","iteration no.",'x1','x2','x3');
                disp([(1:iter)'x']);

                2.gauss jordan method:

                a = [1 1 1; 2 -3 4; 3 4 5];
                b = [9; 13; 40];
                n = length(b);
                aug = [a, b];
                for j = 1:n-1
                    for i = j+1:n
                        aug(i,j:n+1) = aug(i,j:n+1) - aug(i,j)/aug(j,j)*aug(j,j:n+1);
                        disp(aug)
                    end
                end
                for j = n:-1:2
                    aug(1:j-1,:) = aug(1:j-1,:) - aug(1:j-1,j)/aug(j,j)*aug(j,:);
                    disp(aug)
                end
                for j = 1:n
                    aug(j,:) = aug(j,:) / aug(j,j);
                end
                x = aug(:,n+1);
                disp("The solution of the given equations is:");
                disp(strcat(["x = ", string(x(1))]))
                disp(strcat(["y = ", string(x(2))]))

Solutions of differencial equation: 


                1.euler's method:

                function [y0]= euler(x0,y0,h,yest,f)
                    n = (yest-x0)/h
                    for i = 1:n
                        y0 = y0 +f(x0,y0)*h;
                        x0 = x0 +h;
                    end
                endfunction
                deff('[y] = f(x,y)','y = y-x*y+x');
                z = euler(0, 1, 0.2, 1, f);
                mprintf("the answer is : %f",z);

                2.modified euler's method:

                function [y10] = eulermod(x0, y0, h, n, f)
                    x1 = x0 +h;
                    y10 = y0 +h*f(x0, y0);
                    while (n>1)
                        x0 = x0 +h;
                        x1 = y10;
                        y10 = y0 +(h/2)*(f(x0,y0)+f(x0,y10));
                        if(abs(y10-x1)<0.0001)
                            y10
                            abort
                    end
                    n = n-1;
                end
                endfunction
                deff('[y]=f(x,y)','y = log(x+y)');
                z = eulermod(1, 2, 0.2, 10, f);
                printf("answer is %f",z);
